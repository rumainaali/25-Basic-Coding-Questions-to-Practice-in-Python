# Find maximum of two numbers - Way 1
def maxTwoNum(n):
    if len(n)==2:
        if n[0]>n[1]:
            return n[0]
        elif n[1]>n[0]:
            return n[1]
        else:
            return n[0]
    return False

number = maxTwoNum(list(map(int,input().split(" "))))
if number:
    print(f"Maximum of Two numbers: {number}")
else:
    print("Provide two numbers only")




# Find maximum of two numbers - Way 2 (using max())
def maxTwoNum(n):
    if len(n)==2:
        return max(n)
    return False

number = maxTwoNum(list(map(int,input().split(" "))))
if number:
    print(f"Maximum of Two numbers: {number}")
else:
    print("Provide two numbers only")
